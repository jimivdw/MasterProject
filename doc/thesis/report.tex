\newcommand{\reflist}[4]{%
  \getlength{#4}%
  \ifthenelse{\equal{\listlength}{1}}%
    {\archref{#1~\ref{#3:#4}}{#3:#4}}%
    {%
      \def\l{#4}%
      \foreach \e [count=\ei] in \l {%
        \ifthenelse{\equal{\ei}{1}}%
          {#2~\ref{#3:\e}}%
          {%
            \ifthenelse{\equal{\ei}{\listlength}}%
              {%
                \ifthenelse{\equal{\listlength}{2}}%
                  { and \ref{#3:\e}}%
                  {, and \ref{#3:\e}}%
              }%
              {, \ref{#3:\e}}%
          }%
      }%
    }%
}
